- **Slopometer** tells you what's wrong, but you still have to fix it yourself

### The Real Example: Slopometer's Own Security Vulnerability

During Slopometer's security audit, we discovered a **critical vulnerability**:

```typescript
// BEFORE (VULNERABLE)
// src/lib/auth/github.ts
export async function setSession(userId: string, accessToken: string) {
  const cookieStore = await cookies();
  const sessionData = JSON.stringify({ userId, accessToken });
  
  // âŒ CRITICAL: Plain text session with GitHub access token!
  // Anyone who intercepts this cookie gets full GitHub access
  cookieStore.set(SESSION_COOKIE_NAME, sessionData, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 60 * 60 * 24 * 7, // 7 days - too long!
  });
}
```

**What orion.ai would detect and fix:**

1. **Detection:** "Session cookie contains unencrypted GitHub access token"
2. **Risk Score:** CRITICAL (9.8/10)
3. **Auto-Fix Generated:**

```typescript
// AFTER (SECURE) - Generated by orion.ai
import { encrypt, decrypt } from "@/lib/crypto";

export async function setSession(userId: string, accessToken: string) {
  const cookieStore = await cookies();
  const sessionData = JSON.stringify({ userId, accessToken });
  const encrypted = await encrypt(sessionData); // âœ… AES-256-GCM encrypted
  
  cookieStore.set(SESSION_COOKIE_NAME, encrypted, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 60 * 60 * 24, // âœ… 24 hours, not 7 days
  });
}
```

4. **Also Generated:**
   - `src/lib/crypto.ts` â€” Full encryption module
   - `tests/security/session-encryption.test.ts` â€” Security tests
   - `.env.example` update with `ENCRYPTION_SECRET`
   - Documentation for key rotation

### What orion.ai Detects (Examples from Slopometer Audit)

#### 1. Unencrypted Sensitive Data in Cookies

**Detection Pattern:**
```typescript
// orion.ai scanner
function detectUnencryptedCookies(files: RepoFile[]): SecurityFinding[] {
  const findings: SecurityFinding[] = [];
  
  for (const file of files) {
    // Pattern: JSON.stringify with sensitive field names going into cookies
    const pattern = /JSON\.stringify\s*\(\s*\{[^}]*(token|secret|password|key|accessToken)[^}]*\}/gi;
    const cookiePattern = /cookieStore\.set|cookies\(\)\.set|setCookie/gi;
    
    if (pattern.test(file.content) && cookiePattern.test(file.content)) {
      findings.push({
        type: "unencrypted-sensitive-cookie",
        severity: "critical",
        file: file.path,
        message: "Sensitive data stored unencrypted in cookie",
        fix: "encrypt-cookie-data"
      });
    }
  }
  
  return findings;
}
```

**Generated Fix:**
```typescript
// orion.ai generates: src/lib/crypto.ts
import { createCipheriv, createDecipheriv, randomBytes, scryptSync } from "crypto";

const ALGORITHM = "aes-256-gcm";
const IV_LENGTH = 16;
const TAG_LENGTH = 16;
const SALT_LENGTH = 64;
const KEY_LENGTH = 32;

export async function encrypt(text: string): Promise<string> {
  const secret = process.env.ENCRYPTION_SECRET;
  if (!secret) throw new Error("ENCRYPTION_SECRET not set");
  
  const salt = randomBytes(SALT_LENGTH);
  const key = scryptSync(secret, salt, KEY_LENGTH);
  const iv = randomBytes(IV_LENGTH);
  const cipher = createCipheriv(ALGORITHM, key, iv);
  
  let encrypted = cipher.update(text, "utf8", "hex");
  encrypted += cipher.final("hex");
  const tag = cipher.getAuthTag();
  
  return `${salt.toString("hex")}:${iv.toString("hex")}:${encrypted}:${tag.toString("hex")}`;
}

export async function decrypt(encryptedText: string): Promise<string> {
  const secret = process.env.ENCRYPTION_SECRET;
  if (!secret) throw new Error("ENCRYPTION_SECRET not set");
  
  const [saltHex, ivHex, encrypted, tagHex] = encryptedText.split(":");
  const salt = Buffer.from(saltHex, "hex");
  const iv = Buffer.from(ivHex, "hex");
  const tag = Buffer.from(tagHex, "hex");
  const key = scryptSync(secret, salt, KEY_LENGTH);
  
  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(tag);
  
  let decrypted = decipher.update(encrypted, "hex", "utf8");
  decrypted += decipher.final("utf8");
  
  return decrypted;
}
```

#### 3. SSRF Vulnerability in URL Parsing

**Detection Pattern:**
```typescript
// orion.ai scanner
function detectSSRF(files: RepoFile[]): SecurityFinding[] {
  const findings: SecurityFinding[] = [];
  
  for (const file of files) {
    // Pattern: URL parsing without hostname validation
    const urlPattern = /new URL\([^)]+\)/g;
    const fetchPattern = /fetch\([^)]*\)/g;
    
    if (urlPattern.test(file.content) && fetchPattern.test(file.content)) {
      // Check if hostname is validated
      if (!file.content.includes("hostname === ") && 
          !file.content.includes("hostname !== ")) {
        findings.push({
          type: "ssrf-url-validation",
          severity: "high",
          file: file.path,
          message: "URL parsing without hostname validation enables SSRF",
          fix: "validate-url-hostname"
        });
      }
    }
  }
  
  return findings;
}
```

**Generated Fix:**
```typescript
// Before (vulnerable)
export function parseGitHubUrl(url: string): GitHubRepo | null {
  const patterns = [/github\.com\/([^\/]+)\/([^\/\s#?]+)/];
  for (const pattern of patterns) {
    const match = url.match(pattern);
    if (match) return { owner: match[1], name: match[2] };
  }
  return null;
}

// After (secure) - Generated by orion.ai
export function parseGitHubUrl(url: string): GitHubRepo | null {
  try {
    const urlObj = new URL(url);
    // âœ… Explicit hostname check prevents SSRF
    if (urlObj.hostname !== "github.com") {
      return null;
    }
    
    const match = urlObj.pathname.match(/^\/([^\/]+)\/([^\/\s#?]+)/);
    if (match) {
      return { owner: match[1], name: match[2].replace(/\.git$/, "") };
    }
  } catch (e) {
    return null; // Invalid URL
  }
  return null;
}
```

### Generation Engine

```typescript
class GenerationEngine {
  private ai: AIClient;
  
  async generateForCategory(
    category: Category,
    context: RepoContext,
    gaps: Gap[]
  ): Promise<GeneratedFile[]> {
    const files: GeneratedFile[] = [];
    
    // Build context-aware prompt
    const prompt = this.buildPrompt(category, context, gaps);
    
    // Generate with multi-model validation
    const [claudeResult, gptResult] = await Promise.all([
      this.ai.call("claude-sonnet", prompt),
      this.ai.call("gpt-4o", prompt),
    ]);
    
    // Validate and merge results
    const validated = await this.validate(claudeResult, gptResult, context);
    
    // Parse into files
    for (const file of validated.files) {
      files.push({
        path: file.path,
        content: file.content,
        language: detectLanguage(file.path),
        category,
        confidence: file.confidence,
      });
    }
    
    return files;
  }
  
  private async validate(
    claude: AIResponse,
    gpt: AIResponse,
    context: RepoContext
  ): Promise<ValidatedGeneration> {
    // 1. Syntax check
    for (const file of [...claude.files, ...gpt.files]) {
      await this.syntaxCheck(file);
    }
    
    // 2. Import validation
    for (const file of claude.files) {
      await this.validateImports(file, context);
    }
    
    // 3. Type checking (if TypeScript)
    if (context.techStack.languages.includes("typescript")) {
      await this.typeCheck(claude.files);
    }
    
    // 4. Consensus (prefer Claude, fallback to GPT)
    return this.mergeResults(claude, gpt);
  }
}
```

### Test Generation

```typescript
class TestGenerator {
  async generateTests(
    files: RepoFile[],
    coverage: CoverageReport
  ): Promise<GeneratedFile[]> {
    const tests: GeneratedFile[] = [];
    
    // Find untested functions
    const untested = this.findUntested(files, coverage);
    
    for (const fn of untested) {
      const testFile = await this.generateTestForFunction(fn);
      tests.push(testFile);
    }
    
    // Generate integration tests for API routes
    const apiRoutes = files.filter(f => f.path.includes("/api/"));
    for (const route of apiRoutes) {
      const apiTests = await this.generateAPITests(route);
      tests.push(...apiTests);
    }
    
    // Generate E2E tests for critical flows
    const flows = this.detectCriticalFlows(files);
    for (const flow of flows) {
      const e2eTest = await this.generateE2ETest(flow);
      tests.push(e2eTest);
    }
    
    return tests;
  }
  
  private async generateTestForFunction(fn: FunctionDef): Promise<GeneratedFile> {
    const prompt = `
      Generate comprehensive unit tests for this function:
      
      File: ${fn.file}
      Function: ${fn.name}
      Code:
      \`\`\`${fn.language}
      ${fn.code}
      \`\`\`
      
      Requirements:
      - Use ${fn.testFramework || "vitest"}
      - Test happy path
      - Test edge cases (null, undefined, empty)
      - Test error conditions
      - Use descriptive test names
      - Mock external dependencies
    `;
    
    const result = await this.ai.call("claude-sonnet", prompt);
    
    return {
      path: fn.file.replace(/\.(ts|js)$/, ".test.$1"),
      content: result.code,
      language: fn.language,
      category: "testing",
      confidence: result.confidence,
    };
  }
}
```

## Database Schema (Additions to Slopometer)

```sql
-- Completions (orion.ai specific)
CREATE TABLE completions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  scan_id UUID REFERENCES scans(id),  -- Link to Slopometer scan
  
  -- Repo info (duplicated for standalone use)
  repo_url TEXT NOT NULL,
  repo_owner TEXT NOT NULL,
  repo_name TEXT NOT NULL,
  
  -- Completeness analysis
  completeness_score INTEGER NOT NULL,  -- 0-100
  category_scores JSONB NOT NULL,       -- { frontend: 80, backend: 70, ... }
  gaps JSONB NOT NULL,                   -- Detailed gaps by category
  
  -- Generation
  files_generated INTEGER DEFAULT 0,
  generation_status TEXT DEFAULT 'pending',
  
  -- Output
  pr_url TEXT,
  zip_url TEXT,
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ
);

-- Generated files
CREATE TABLE generated_files (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  completion_id UUID REFERENCES completions(id) ON DELETE CASCADE,
  
  file_path TEXT NOT NULL,
  file_content TEXT NOT NULL,
  language TEXT NOT NULL,
  category TEXT NOT NULL,
  
  -- Quality
  confidence INTEGER NOT NULL,
  validated BOOLEAN DEFAULT false,
  validation_errors JSONB,
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Generation credits
CREATE TABLE generation_credits (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  
  credits_used INTEGER NOT NULL,
  generation_type TEXT NOT NULL,  -- 'single', 'category', 'full'
  completion_id UUID REFERENCES completions(id),
  
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes
CREATE INDEX idx_completions_user ON completions(user_id);
CREATE INDEX idx_completions_scan ON completions(scan_id);
CREATE INDEX idx_generated_files_completion ON generated_files(completion_id);
CREATE INDEX idx_credits_user ON generation_credits(user_id);
```
